#### useState

```javascript
import { useState } from "react"
function App(){
    //constent是本次渲染的数据内容，setConstent是用来修改状态的函数
    const [constent,setConstent] = useState('默认内容')
    function handleClick (){
        setConstent('修改内容')
    }
    const [data,setData] = useState({
        title: '默认标题',
        content: '默认内容'
    })
    function handleClick (){
        setConstent('修改内容')
    }
    function handleClick1 (){
        //由于这个函数的逻辑是直接替换状态，所以要重新写一个对象,数组同理
        setConstent({
            ...data,
            content: '新内容'
        })
    }
    return(
    	<>
        	<div>{constent}</div>
        	<button onClick={handleClick}>按钮</button>
			<div>{data.content}</div>
        	<button onClick={handleClick1}>按钮</button>
        </>
    )
}
```
